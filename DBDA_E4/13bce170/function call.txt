 CREATE OR REPLACE FUNCTION user_info_fun()
     RETURN number(10)
     IS
         user number(10) := 0;
     BEGIN
       SELECT  count(*) into user FROM user_info;
        RETURN user;
    END;
 /


 DECLARE
 c number(20);
 BEGIN
 c := user_info_fun();
 dbms_output.put_line('total number of user  :' || c);
 END;
 /
