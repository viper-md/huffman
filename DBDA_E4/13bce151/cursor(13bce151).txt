declare
pro_rno student.rollno%type;
pro_fname student.fname%type;
pro_lname student.lname%type;
CURSOR c_pro is
select rollno,fname,lname from student;
begin
open c_pro;
loop
fetch c_pro into pro_rno,pro_fname,pro_lname;
exit when c_pro%notfound;
dbms_output.put_line(pro_rno ||' || '|| pro_fname ||' || '|| pro_lname);
end loop;
close c_pro;
end;
